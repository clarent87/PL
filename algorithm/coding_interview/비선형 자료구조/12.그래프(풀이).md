```python
from typing import List


class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        def dfs(i, j):
            # 더 이상 땅이 아닌 경우 종료
            if i < 0 or i >= len(grid) or \
                    j < 0 or j >= len(grid[0]) or \
                    grid[i][j] != '1':
                return

            grid[i][j] = 0 # 방문한곳을 0으로 만듬. 일종의 가지치기

            # 동서남북 탐색
            dfs(i + 1, j)
            dfs(i - 1, j)
            dfs(i, j + 1)
            dfs(i, j - 1)

        count = 0
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j] == '1':
                    dfs(i, j)
                    # 모든 육지 탐색 후 카운트 1 증가
                    count += 1
        return count

```

- 위 풀이 포인트

    - > 아 ㅋㅋ 336 읽으면 이해 된다.
      - 즉, "부모 변수 = 값" 을하면 함수내의 로컬 변수를 할당하는 것으로 인식. 즉 로컬변수가 생김
      - 이게 싫으면 `nonlocal 부모변수 = 값`을 해주면 부모변수에 할당 가능
    - 특징
      - 부모 함수의 변수를 자유롭게 읽을수 있음

  ```python
    def combine(self, n: int, k: int) -> List[List[int]]:
        results = []

        # start 값에서부터 내부적으로 n+1까지의 vertex를 만듬.
        # vertex는 1,2,3,.. 임
        def dfs(elements, start: int, k: int):
            if k == 0:
                results.append(elements[:]) # 백트래킹이라고 표현해야할까? 맞는듯. DFS는 백트래킹 알고리즘

            for i in range(start, n + 1): # 1부터 값을 만드니까 n+1
                elements.append(i)
                dfs(elements, i + 1, k - 1)
                elements.pop()

        dfs([], 1, k) 
        return results
  ```

```python
  def combinationSum(self, candidates: List[int], target: int) -> List[List[int]]:
      result = []

      def dfs(csum, index, path):
          # 종료 조건
          if csum < 0:
              return
          if csum == 0:
              result.append(path)
              return

          # 자신 부터 하위 원소 까지의 나열 재귀 호출
          for i in range(index, len(candidates)): # 잘 보면 range는 index ~ candidates길이까지임. 즉 "전화번호 문제"유사
              dfs(csum - candidates[i], i, path + [candidates[i]]) # 두번째 param인 i를 0으로 고정하면 순열이 됨

      dfs(target, 0, [])
      return result
```

  ```python
      def findItinerary(self, tickets: List[List[str]]) -> List[str]:
          graph = collections.defaultdict(list)
          # 그래프 순서대로 구성
          for a, b in sorted(tickets):
              graph[a].append(b)

          route, stack = [], ['JFK']

          while stack:
              # 반복으로 스택을 구성하되 막히는 부분에서 풀어내는 처리
              while graph[stack[-1]]:
                  stack.append(graph[stack[-1]].pop(0))
              route.append(stack.pop()) # 요 구문을 362에서 중점으로 설명하네..  stack.pop()

          # 다시 뒤집어 어휘순 결과로
          return route[::-1]
  ```

## 207. 코스 스케줄

- 문제 이해는 리트코드 보는게 수월하다.
- 그래프가 cycle인지 판별하는 문제.
  - > 순환은 depth first 에서 한큐에 순환이 생겨함. 백트랙으로 경로를 거쳤다고 순환은 아님

- 풀이1 . DFS로 순환 구조 판별
  - > 역시 전통의 DFS랑은 살짝 다른듯 (사실 거의 비슷)
  - > 여전히 시작 vertex는 여러개인 거고.
  - > 사실 종료 조건이 기본 DFS랑 다르긴해서... 그런듯

```python

  numCourses = 2, prerequisites = [[1,0],[0,1]]

  def canFinish(self, numCourses: int, prerequisites: List[List[int]]) -> bool:
      graph = collections.defaultdict(list)
      
      # 그래프 구성 -> 역시 인접 리스트로 구현
      for x, y in prerequisites:
          graph[x].append(y) 

      traced = set() # 여기서는 list가 아닌 set을 썻네.. list 써도 상관은 없을듯.
                    # 오해하면 안되는게 traced는 discovered가 아님. 즉 방문 vertex를 의미하는게 아님
                    # 그냥 cycle을 판별하기 위한것. 그래서 dfs 끝날땐 항상 값을 뺌

      def dfs(i):
          # 순환 구조이면 False
          if i in traced:
              return False

          traced.add(i)
          for y in graph[i]:
              if not dfs(y):
                  return False
          # 탐색 종료 후 순환 노드 삭제
          # 이쪽 path는 순환이 없다는것. (trace에서 빼줘야 한다 다음 vertex를 위해서)
          traced.remove(i)

          return True

      # 순환 구조 판별
      for x in list(graph): # dict는 원래 key가 빠짐. list 사실 안붙여도 된다. 
          if not dfs(x):
              return False

      return True

```

- 풀이2. 가지치기를 이용한 최적화
  - DFS는 원래 이렇게 가지치기를 하는게 올바른 방법

```python
  def canFinish(self, numCourses: int, prerequisites: List[List[int]]) -> bool:
      graph = collections.defaultdict(list)
      # 그래프 구성
      for x, y in prerequisites:
          graph[x].append(y)

      traced = set()
      visited = set()

      def dfs(i):
          # 순환 구조이면 False
          if i in traced:
              return False
          # 이미 방문했던 노드이면 True
          if i in visited:
              return True

          traced.add(i)
          for y in graph[i]:
              if not dfs(y):
                  return False
          # 탐색 종료 후 순환 노드 삭제
          traced.remove(i)
          # 탐색 종료 후 방문 노드 추가
          visited.add(i)

          return True

      # 순환 구조 판별
      for x in list(graph):
          if not dfs(x):
              return False

      return True

```
